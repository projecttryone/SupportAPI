def esc(s: str) -> str:
    return s.replace("&", "&amp;").replace("<", "&lt;").replace(">", "&gt;")

def rich(s: str) -> str:
    """Escape, then restore a tiny markup set: **bold**, `code`, [[term]] accent."""
    s = esc(s)

    def _code(m):
        inner = m.group(1).replace("*", "&#42;")  # keep bold pass out of code spans
        return f'<font face="Courier" size="8.6" color="#334155">{inner}</font>'

    s = re.sub(r"`([^`]+)`", _code, s)
    s = re.sub(r"\*\*(.+?)\*\*", r"<b>\1</b>", s)
    s = re.sub(r"\[\[(.+?)\]\]", r'<b><font color="#c2410c">\1</font></b>', s)
    return s

# ---------------------------------------------------------------- flowable builders
def code_block(body: str, title: str | None = None):
    pre = Preformatted(esc(body.rstrip("\n")), S["code"])
    t = Table([[pre]], colWidths=[CW])
    t.setStyle(TableStyle([
        ("BACKGROUND", (0, 0), (-1, -1), CODE_BG),
        ("BOX", (0, 0), (-1, -1), 0.6, CODE_BD),
        ("LINEBEFORE", (0, 0), (0, -1), 2.4, NAVY),
        ("LEFTPADDING", (0, 0), (-1, -1), 10),
        ("RIGHTPADDING", (0, 0), (-1, -1), 8),
        ("TOPPADDING", (0, 0), (-1, -1), 6),
        ("BOTTOMPADDING", (0, 0), (-1, -1), 6),
    ]))
    out = []
    if title:
        out.append(Paragraph(rich(title), S["codetitle"]))
    out.append(t)
    out.append(Spacer(1, 6))
    return out

CALLOUT_KIND = {
    "bug":   ("BUG",        BUG_BG, BUG_FG),
    "fix":   ("THE FIX",    FIX_BG, FIX_FG),

def callout(kind: str, title: str, body: str):
    label, bg, fg = CALLOUT_KIND[kind]
    tstyle = ParagraphStyle(f"ct_{kind}", parent=S["callout_t"], textColor=fg)
    head = Paragraph(f"{label}{' - ' if title else ''}{rich(title)}", tstyle)
    para = Paragraph(rich(body), S["callout_p"])
    t = Table([[head], [para]], colWidths=[CW])
    t.setStyle(TableStyle([
        ("BACKGROUND", (0, 0), (-1, -1), bg),
        ("LINEBEFORE", (0, 0), (0, -1), 2.4, fg),
        ("LEFTPADDING", (0, 0), (-1, -1), 10),
        ("RIGHTPADDING", (0, 0), (-1, -1), 8),
        ("TOPPADDING", (0, 0), (0, 0), 6),
        ("BOTTOMPADDING", (0, 0), (0, 0), 1),
        ("TOPPADDING", (0, 1), (0, 1), 1),
        ("BOTTOMPADDING", (0, 1), (0, 1), 6),
    ]))
    return [KeepTogether([t]), Spacer(1, 7)]

def table2(headers, rows, widths):
    data = [[Paragraph(rich(h), S["th"]) for h in headers]]
    for r in rows:
        data.append([Paragraph(rich(c), S["td"]) for c in r])
    t = Table(data, colWidths=widths, repeatRows=1)
    style = [
        ("BACKGROUND", (0, 0), (-1, 0), NAVY),
        ("GRID", (0, 0), (-1, -1), 0.5, CODE_BD),
        ("VALIGN", (0, 0), (-1, -1), "TOP"),
        ("LEFTPADDING", (0, 0), (-1, -1), 6),
        ("RIGHTPADDING", (0, 0), (-1, -1), 6),
        ("TOPPADDING", (0, 0), (-1, -1), 4),
        ("BOTTOMPADDING", (0, 0), (-1, -1), 4),
    ]
    for i in range(1, len(data)):
        if i % 2 == 0:
            style.append(("BACKGROUND", (0, i), (-1, i), ROW_ALT))
    t.setStyle(TableStyle(style))
    return [t, Spacer(1, 8)]

h2("Generator functions and yield")
code('''def stream_tickets(path=None) -> Iterator[TicketDict]:
    target = path or (_DATA_DIR / "tickets.json")
    with target.open(encoding="utf-8") as fh:

            yield ticket          # pause here; resume on the next request''',
     "stream.py - your generator")
p("`yield` turns a function into a **generator function**. Calling it runs *no body code* - "
  "it returns a [[generator object]] (an iterator). Each `next(gen)` (or loop iteration) runs "
  "until the next `yield`, hands back one value, and **freezes all local state** until asked "
  "again. That on-demand production is [[lazy evaluation]] - say the term.")
bullets([
    "`next(stream)` pulls exactly one ticket; a `for` loop pulls until exhaustion; a finished "
    "generator raises [[StopIteration]] (the for-loop absorbs it silently).",
    "Generators are [[single-pass]]: once exhausted, they're done. Re-reading means calling "
    "`stream_tickets()` again for a fresh generator - a favorite trick question.",
    "Why it matters: `json.load()` on 5M tickets stalls or crashes; a generator pipeline holds "
    "**one record at a time**.",
])
box("say", "The nuance that separates you from the cohort",
    "Your own docstring already flags it: this generator still calls `json.load(fh)`, which "
    "**parses the whole file into memory first** - the laziness applies to everything "
    "*downstream*, not to the file read. True incremental parsing needs a streaming parser "
    "like `ijson`, or JSON-Lines with one object per line. Volunteering that honest limitation "
    "plus its fix is a senior-signal moment in a mock interview.")

h2("The pipeline: where -> tag -> take")
code('''def where(tickets, predicate):            # FILTER the stream (generator)
    for ticket in tickets:
        if predicate(ticket):
            yield ticket

def tag(tickets, field, compute):         # ENRICH without mutating (generator)
    for ticket in tickets:
        yield {**ticket, field: compute(ticket)}     # new dict via unpacking

def take(tickets, n):                     # MATERIALIZE - the pipeline's end
    result: list[TicketDict] = []
    for ticket in tickets:
        if len(result) >= n:
            break
        result.append(ticket)
    return result                         # return, not yield - eager on purpose''',
     "stream.py (with Part 7's fixes applied)")
code('''stream  = stream_tickets()
high_b  = where(stream, lambda t: t["priority"] == "high" and t["category"] == "billing")
flagged = tag(high_b, "needs_review", lambda t: t["status"] in {"open", "in_progress"})
five    = take(flagged, 5)      # nothing upstream ran until THIS line pulled''',
     "Composed - one ticket in flight at any moment")
bullets([
    "`where` and `tag` use `yield`, so each is itself a generator - that's what makes them "
    "**composable** stages. `take` uses `return` because someone has to end the laziness and "
    "[[materialize]] concrete output.",
    "`{**ticket, field: value}` is [[dictionary unpacking]]: copy every original key, add one "
    "computed field, **never mutate the input**. Non-mutating enrichment keeps upstream data "
    "trustworthy - the same discipline your async `enrich_ticket` follows.",
    "A `predicate` (another term to say out loud) is any function returning a bool - the "
    "`Callable[[dict], bool]` contract in your type hints.",
])
box("agent", "Context bloat - the Week 5 echo",
    "Dumping 500 tickets into an agent's prompt degrades its reasoning the same way "
    "`json.load` on everything degrades memory. Streaming one chunk at a time is both the "
    "memory answer *and* the context-management answer. One pattern, two payoffs - this "
    "mapping is explicitly in the curriculum, so expect it as an exam question.")

h2("Higher-order functions, closures, and the factory")
code('''def matches(predicate: Callable[[TicketDict], bool]
            ) -> Callable[[list[TicketDict]], list[TicketDict]]:
    """Return a filter function scoped to one predicate."""
    def apply(tickets: list[TicketDict]) -> list[TicketDict]:
        return [t for t in tickets if predicate(t)]
    return apply                     # apply REMEMBERS predicate: a closure

is_urgent       = matches(lambda t: t["priority"] == "urgent")
is_billing      = matches(lambda t: t["category"] == "billing")
is_high_billing = matches(lambda t: t["priority"] == "high"
                          and t["category"] == "billing")''',
     "filters.py - matches()")
p("Three terms live in this one function - be able to point at each:")
bullets([
    "[[Higher-order function]]: takes a function as input, returns a function as output - "
    "`matches` does both.",
    "[[Closure]]: the inner `apply` captures `predicate` from its enclosing scope and "
    "**remembers it after `matches` has returned**. Each call to `matches` mints a new "
    "`apply` with its own captured predicate.",
    "[[Factory pattern]] applied to functions: `matches` manufactures configured filters on "
    "demand. Works because Python functions are [[first-class]] - passable, returnable, "
    "assignable like any value.",
])
p("**Lambda rules**: a lambda is a one-off anonymous function limited to a **single "
  "expression** - no statements, no annotations. Perfect as a throwaway predicate argument; "
  "the moment logic needs two steps or a name for reuse, promote it to a `def`.")

h2("Decorators - the three-layer anatomy")
code('''def timed(label: str | None = None):            # 1. FACTORY: takes config
    def decorator(func):                        # 2. DECORATOR: takes the function
        event_label = label or func.__qualname__

        @wraps(func)                            # preserve name + docstring
        def wrapper(*args, **kwargs):           # 3. WRAPPER: replaces the function
            started = time.perf_counter()
            try:
                return func(*args, **kwargs)    # forward everything unchanged
            finally:                            # logs even if func raises
                duration_ms = round((time.perf_counter() - started) * 1000, 2)
                _log.info("timed", label=event_label, duration_ms=duration_ms)
        return wrapper
    return decorator

@timed()                       # parentheses REQUIRED - timed is a factory
def route_ticket(ticket): ...
@timed(label="enrich-batch-run")
def run(): ...''',
     "utils.py - your @timed, and it's better than the course baseline")
bullets([
    "Layer 1 makes it a [[decorator factory]] - configurable at the call site "
    "(`@timed(label=...)`), which is exactly why the bare `@timed` without parentheses "
    "breaks: you'd be decorating with the factory itself.",
    "`*args, **kwargs` forwards any signature unchanged; `return func(...)` preserves the "
    "return value - a decorator must be behavior-transparent (your "
    "`test_timed_returns_func_result` asserts precisely this).",
    "`@functools.wraps(func)` copies `__name__`, `__qualname__`, `__doc__` onto the wrapper. "
    "Without it every decorated function reports as \"wrapper\" - breaking debugging, and "
    "breaking **agents that introspect signatures**.",
    "**Your upgrade worth bragging about**: the `try/finally` means duration logs even when "
    "the function *raises* - the naive version silently loses timing on failures. Cross-"
    "cutting concerns (the term: [[cross-cutting concern]]) - timing, logging, retries, "
    "auth - all follow this shape.",

h2("Context managers - guaranteed setup and teardown")
p("A context manager wraps a block with setup/teardown that runs **even on exceptions**. "
  "You've used one all week: `with target.open() as fh:` closes the file no matter what. Two "
  "ways to build your own: a class with `__enter__`/`__exit__`, or the shortcut you chose - "
  "`@contextlib.contextmanager` on a generator, where `yield` marks the handoff:")
code('''@contextmanager
def atomic_write(path: Path, encoding: str = "utf-8"):
    """Write to a temp file, then atomically rename over the target on success."""
    tmp = path.with_suffix(path.suffix + ".tmp")     # demo.txt -> demo.txt.tmp
    try:
        with tmp.open("w", encoding=encoding) as fh:
            yield fh                  # <- caller writes here (the with-block body)
    except BaseException:             # ANY failure in the block...
        if tmp.exists():
            tmp.unlink()              # ...remove the temp file
        raise                         # ...and re-raise: never swallow
    else:                             # only on a clean exit:
        os.replace(tmp, path)         # ATOMIC swap - the one-line crash-safety win''',
     "utils.py - atomic_write, your strongest interview artifact")
p("Why this is the piece to lead with when asked \"show me something you're proud of\": "

   "what's in target afterward?",
   "The [[original content, untouched]] - the except path unlinks the temp file and "
   "re-raises; `os.replace` only runs in the `else` (clean-exit) branch. That's the whole "
   "point of atomic writes.")
qa("Why does @timed (no parentheses) break while @timed() works?",
   "`timed` is a [[decorator factory]] - it returns the decorator. Without parentheses "
   "you decorate with the factory itself, so calling the function returns the inner "
   "`decorator` instead of running your code.")

h2("Design questions - the \"why\" layer")
qa("Why validate with Pydantic instead of checking fields in the router?",
   "[[Separation of boundary and logic]]: validation happens once, declaratively, at the "
   "edge - so every downstream component trusts its input. Scattered checks drift and get "
   "skipped; a model is a single typed contract, and it's the anti-injection choke point "
   "for agent inputs.")
qa("Why is RoutingDecision frozen?",
   "A decision is a [[fact, not a workspace]] - immutability prevents downstream mutation "
   "bugs, makes it hashable for dedup/caching, and documents intent in one flag.")
qa("Why ordered rules instead of ML from day one?",
   "[[Deterministic baseline first]]: explainable, testable, zero-cost. When the Week 3 "
   "LLM classifier arrives, this engine becomes the eval baseline I diff it against - you "
   "can't measure a model without a ground truth to compare.")
qa("Why does take() use return while where() uses yield?",
   "Pipelines need [[lazy stages and one eager terminus]]. Stages yield to stay "
   "composable; the terminus materializes results. Making take a generator would mean "
   "nothing ever executes.")
qa("Your enrichment is async with a semaphore - why both?",
   "`asyncio.gather` gives [[concurrency]] for I/O-bound HTTP calls; the "
   "[[semaphore]] caps in-flight requests so I respect rate limits and don't stampede the "
   "API - concurrency with a governor, which is exactly how parallel LLM tool calls run.")

p("**Situation**: My ticket router passed a smoke test - every ticket got a queue - but "
  "every single one landed in `finance-lead`. **Task**: Find why five ordered rules "
  "behaved like one. **Action**: I isolated `_rule_matches` and walked it line by line. "
  "Three compounding issues: a `return True` *inside* the field loop, so at most one field "
  "was ever checked; an identity test (`is not`) where membership (`not in`) belonged, so "
  "tuple-valued rules could never match; and the equality `elif` nested under the "
  "`isinstance` branch, so plain string rules were never compared at all - the function "
  "returned True unconditionally and rule one always fired. I restructured it: tuple branch "
  "uses `not in`, the `elif` handles scalars, and the success `return True` moved *after* "
  "the loop. **Result**: All five default rules routed correctly across the 50-ticket seed, "
  "my parametrized pytest cases went green, and I added a regression test asserting an "
  "unknown category raises `NoMatchingRule` - so the failure mode is now a contract, not a "
  "surprise. Lesson I keep: `assert all-fields-matched` logic must return success **after** "
  "the loop, and `is` is for identity, never values.")

# --- CONTEXT MANAGER -----------------------------------------------------
@contextmanager
def atomic_write(path):
    tmp = path.with_suffix(path.suffix + ".tmp")
    try:
        with tmp.open("w") as fh: yield fh  # handoff point
    except BaseException:
        tmp.unlink(missing_ok=True); raise  # cleanup, never swallow
